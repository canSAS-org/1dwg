Function XML_tag(tag, attr, content) As String  If attr = "" Then    XML = "<" & tag & ">"  Else    XML = "<" & tag & " " & attr & ">"  End If  XML = XML & content  XML = XML & "</" & tag & ">"  XML_tag = XMLEnd FunctionFunction SAS_Idata_tag(element, unit, content) As String  XML = XML_tag(element, "unit=""" & unit & """", content)  SAS_Idata_tag = XMLEnd FunctionFunction Idata_tag(Q, Q_unit, I, I_unit, Qdev, Qdev_unit, Idev, Idev_unit) As String  XML = SAS_Idata_tag("Q", Q_unit, Q)  XML = XML & SAS_Idata_tag("I", I_unit, I)  XML = XML & SAS_Idata_tag("Qdev", Qdev_unit, Qdev)  XML = XML & SAS_Idata_tag("Idev", Idev_unit, Idev)  Idata_tag = XML_tag("Idata", "", XML)End Function
